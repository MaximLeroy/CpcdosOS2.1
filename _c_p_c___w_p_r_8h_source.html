137  void CpcdosOSx_CPintiCore::fdebug_log(long alert, const char* format, ...)
138  {
139  // Cette fonction permet d'utiliser le simuler un sprintf()
140  va_list arg;
141  char BUFFER[1024] = {0};
142 
143  va_start (arg, format);
144  vsprintf (BUFFER, format, arg);
145  va_end (arg);
146 
147 
148  cpc_cpinti_debug_cpinticore(BUFFER, alert);
149 
150  // Remplir le buffer de ZEROS
151  memset(BUFFER, '\0', sizeof(char)*1024);
152 
153  }
